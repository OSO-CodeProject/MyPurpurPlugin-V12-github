package org.example;

import net.kyori.adventure.text.Component;
import net.kyori.adventure.text.format.NamedTextColor;
import org.bukkit.Material;

import org.bukkit.entity.Player;

import org.bukkit.command.Command;

import org.bukkit.plugin.java.JavaPlugin;
import org.jetbrains.annotations.NotNull;

/**
 * Обработчик команды /menu для открытия меню игрокам.
 */
public class MenuCommand implements org.bukkit.command.CommandExecutor {

    private final JavaPlugin plugin;
    private final PluginConfig pluginConfig;

    public MenuCommand(@NotNull JavaPlugin plugin, @NotNull PluginConfig pluginConfig) {
        this.plugin = plugin;
        this.pluginConfig = pluginConfig;
    }

    @Override
    public boolean onCommand(@NotNull CommandSender sender, @NotNull Command command, @NotNull String label, @NotNull String[] args) {
        if (!(sender instanceof Player player)) {
            sender.sendMessage(Component.text("❌ Эту команду может использовать только игрок!", NamedTextColor.RED));
            return true;
        }
        openMenu(player);
        return true;
    }

    private void openMenu(@NotNull Player player) {
        Inventory menu = plugin.getServer().createInventory(null, 9, Component.text("Меню"));

        ItemStack exampleItem = new ItemStack(Material.DIAMOND);
        ItemMeta meta = exampleItem.getItemMeta();
        if (meta != null) {
            meta.displayName(Component.text("Пример предмета"));
            exampleItem.setItemMeta(meta);
        }

        menu.setItem(4, exampleItem);

        player.openInventory(menu);
        player.playSound(
            player.getLocation(),
            Sound.valueOf(pluginConfig.getMenuOpenSound().toUpperCase()),
            Sound.Source.MASTER,
            (float) pluginConfig.getMenuSoundVolume(),
            (float) pluginConfig.getMenuSoundPitch()
        );
        player.getWorld().spawnParticle(org.bukkit.Particle.valueOf(pluginConfig.getMenuParticleEffect().toUpperCase()), player.getLocation().add(0.5, 1.0, 0.5), 30, 0.5, 0.5, 0.5, 0.1);
    }
}

        String subCommand = args[0].toLowerCase();
        ((MyPurpurPlugin) teamManager.getPlugin()).debugTeamAction("Команда /team выполнена игроком", player.getName(), null);

        return handleSubCommand(player, subCommand, args);

    private boolean handleSubCommand(Player player, String subCommand, String[] args) {
        return switch (subCommand) {
            case "create" -> handleCreateCommand(player, args);
            case "join" -> handleJoinCommand(player, args);
            case "leave" -> handleLeaveCommand(player);
            case "transfer" -> handleTransferCommand(player, args);
            case "kick" -> handleKickCommand(player, args);
            case "disband" -> handleDisbandCommand(player);
            case "list" -> handleListCommand(player);
            case "members" -> handleMembersCommand(player);
            case "rename" -> handleRenameCommand(player, args);
            case "setprefix" -> handleSetPrefixCommand(player, args);
            case "setcolor" -> handleSetColorCommand(player, args);
            case "help" -> {
                sendHelp(player);
                yield true;
            }
            default -> {
                player.sendMessage(Component.text("❌ Неизвестная подкоманда! Используйте: /team <create|join|leave|transfer|kick|disband|list|members|rename|setprefix|setcolor|help>", NamedTextColor.RED));

/**
 * Реализация сервиса управления командами.
 */
public class TeamManager implements TeamService {

    private static final String TEAMS_PATH = "teams.";
    private static final String LEADER_PATH = ".leader";
    private static final String MEMBERS_PATH = ".members";
    private static final String PREFIX_PATH = ".prefix";
    private static final String COLOR_PATH = ".color";

    private final JavaPlugin plugin;
    private FileConfiguration teamsConfig;
    private File teamsFile;
    private final Map<String, List<String>> teamMembersCache;
    private final Map<String, String> teamPrefixCache;
    private final Map<String, NamedTextColor> teamColorCache;

    public TeamManager(@NotNull JavaPlugin plugin) {
        this.plugin = plugin;
        this.teamMembersCache = new HashMap<>();
        this.teamPrefixCache = new HashMap<>();
        this.teamColorCache = new HashMap<>();
        loadTeams();
    }

    private void loadTeams() {
        teamsFile = new File(plugin.getDataFolder(), "teams.yml");

        // Убеждаемся, что папка плагина существует
        if (!plugin.getDataFolder().exists()) {
            boolean mkdirsResult = plugin.getDataFolder().mkdirs();
            if (!mkdirsResult) {
                plugin.getLogger().severe("⚠ Не удалось создать папку плагина: " + plugin.getDataFolder().getAbsolutePath());
                return;
            }
            plugin.debug("📂 Папка плагина создана: " + plugin.getDataFolder().getAbsolutePath());
        }

        if (!teamsFile.exists()) {
            try {
                if (teamsFile.createNewFile()) {
                    plugin.getLogger().info("📂 Файл teams.yml успешно создан!");
                }
            } catch (IOException e) {
                plugin.getLogger().severe("⚠ Ошибка создания teams.yml: " + e.getMessage());
            }
        }

        teamsConfig = YamlConfiguration.loadConfiguration(teamsFile);
        reloadCaches();
        plugin.debug("📂 Файл teams.yml загружен!");
    }

    private void saveTeams() {
        try {
            teamsConfig.save(teamsFile);
            reloadCaches(); // Обновляем кэш после сохранения
            plugin.debug("📂 Файл teams.yml успешно сохранён.");
        } catch (IOException e) {
            plugin.getLogger().severe("⚠ Ошибка сохранения teams.yml: " + e.getMessage());
        }
    }

    private void reloadCaches() {
        teamMembersCache.clear();
        teamPrefixCache.clear();
        teamColorCache.clear();

        List<String> teamNames = getTeamNamesFromConfig();
        for (String teamName : teamNames) {
            teamMembersCache.put(teamName, teamsConfig.getStringList(TEAMS_PATH + teamName + MEMBERS_PATH));
            teamPrefixCache.put(teamName, teamsConfig.getString(TEAMS_PATH + teamName + PREFIX_PATH, ""));
            String color = teamsConfig.getString(TEAMS_PATH + teamName + COLOR_PATH, "WHITE");
            teamColorCache.put(teamName, NamedTextColor.NAMES.value(color.toLowerCase()));
        }
    }

    private List<String> getTeamNamesFromConfig() {
        if (teamsConfig.getConfigurationSection(TEAMS_PATH) == null) {
            return new ArrayList<>();
        }
        return new ArrayList<>(teamsConfig.getConfigurationSection(TEAMS_PATH).getKeys(false));
    }

    @Override
    public void createTeam(String teamName, String prefix, String color, @NotNull Player leader) {
        ((MyPurpurPlugin) plugin).debugTeamAction("Попытка создания команды", leader.getName(), teamName);
        String existingTeam = getPlayerTeam(leader);
        if (existingTeam != null) {
            TeamMessageUtils.sendTeamMessage(leader, Component.text("❌ Вы не можете создать команду, так как уже состоите в команде " + existingTeam + "!", NamedTextColor.RED));
            return;
        }

        if (teamsConfig.contains(TEAMS_PATH + teamName)) {
            TeamMessageUtils.sendTeamMessage(leader, Component.text("❌ Команда уже существует!", NamedTextColor.RED));
            return;
        }

        for (String existingTeamName : getTeamNames()) {
            String existingPrefix = getTeamPrefix(existingTeamName);
            if (prefix.equals(existingPrefix)) {
                TeamMessageUtils.sendTeamMessage(leader, Component.text("❌ Префикс '" + prefix + "' уже используется другой командой!", NamedTextColor.RED));
                return;
            }
        }

        NamedTextColor teamColor;
        try {
            teamColor = NamedTextColor.NAMES.value(color.toLowerCase());
            if (teamColor == null) {
                TeamMessageUtils.sendTeamMessage(leader, Component.text("❌ Неверный цвет команды. Используйте, например: RED, BLUE, GREEN и т.д.", NamedTextColor.RED));
                return;
            }
        } catch (IllegalArgumentException e) {
            TeamMessageUtils.sendTeamMessage(leader, Component.text("❌ Неверный цвет команды. Используйте, например: RED, BLUE, GREEN и т.д.", NamedTextColor.RED));
            return;
        }

        for (String existingTeamName : getTeamNames()) {
            NamedTextColor existingColor = getTeamColor(existingTeamName);
            if (color.equalsIgnoreCase(existingColor.toString())) {
                TeamMessageUtils.sendTeamMessage(leader, Component.text("❌ Цвет '" + color + "' уже используется другой командой!", NamedTextColor.RED));
                return;
            }
        }

        teamsConfig.set(TEAMS_PATH + teamName + LEADER_PATH, leader.getName());
        teamsConfig.set(TEAMS_PATH + teamName + MEMBERS_PATH, List.of(leader.getName()));
        teamsConfig.set(TEAMS_PATH + teamName + PREFIX_PATH, prefix);
        teamsConfig.set(TEAMS_PATH + teamName + COLOR_PATH, color.toUpperCase());
        saveTeams();

        Component prefixComponent = TeamUtils.createPrefixComponent(prefix, teamColor);
        Component message = Component.text("✅ Команда ")
                .append(prefixComponent)
                .append(Component.text("[" + teamName + "] создана! Вы теперь лидер команды "))
                .append(prefixComponent)
                .append(Component.text("[" + teamName + "]!", NamedTextColor.GREEN));
        TeamMessageUtils.sendTeamMessage(leader, message);
        updatePlayerPrefixes(teamName);
        plugin.getLogger().info("Команда " + teamName + " успешно создана лидером " + leader.getName());
    }

    @Override
    public void addPlayerToTeam(String teamName, @NotNull Player player) {
        ((MyPurpurPlugin) plugin).debugTeamAction("Попытка добавления игрока в команду", player.getName(), teamName);
        if (!teamsConfig.contains(TEAMS_PATH + teamName)) {
            TeamMessageUtils.sendTeamMessage(player, Component.text("❌ Такой команды не существует!", NamedTextColor.RED));
            return;
        }

        String existingTeam = getPlayerTeam(player);
        if (existingTeam != null) {
            TeamMessageUtils.sendTeamMessage(player, Component.text("❌ Вы уже состоите в команде " + existingTeam + "!", NamedTextColor.RED));
            return;
        }

        for (String team : getTeamNames()) {
            String leader = getTeamLeader(team);
            if (player.getName().equals(leader)) {
                TeamMessageUtils.sendTeamMessage(player, Component.text("❌ Вы не можете вступить в другую команду, так как являетесь лидером команды " + team + "!", NamedTextColor.RED));
                return;
            }
        }

        List<String> members = getTeamMembers(teamName);
        if (members.contains(player.getName())) {
            TeamMessageUtils.sendTeamMessage(player, Component.text("❌ Вы уже в этой команде!", NamedTextColor.RED));
            return;
        }

        members.add(player.getName());
        teamsConfig.set(TEAMS_PATH + teamName + MEMBERS_PATH, members);
        saveTeams();

        String prefix = getTeamPrefix(teamName);
        NamedTextColor teamColor = getTeamColor(teamName);
        Component prefixComponent = TeamUtils.createPrefixComponent(prefix, teamColor);

        for (String memberName : members) {
            Player member = plugin.getServer().getPlayer(memberName);
            if (member != null && !member.equals(player)) {
                Component message = Component.text("✅ Игрок " + player.getName() + " вступил в команду ")
                        .append(prefixComponent)
                        .append(Component.text("[" + teamName + "]!", NamedTextColor.GREEN));
                TeamMessageUtils.sendTeamMessage(member, message);
            }
        }
        Component playerMessage = Component.text("✅ Вы вступили в команду ")
                .append(prefixComponent)
                .append(Component.text("[" + teamName + "]!", NamedTextColor.GREEN));
        TeamMessageUtils.sendTeamMessage(player, playerMessage);
        updatePlayerPrefixes(teamName);
        plugin.getLogger().info("Игрок " + player.getName() + " вступил в команду " + teamName);
    }

    @Override
    public void removePlayerFromTeam(String teamName, @NotNull Player player) {
        ((MyPurpurPlugin) plugin).debugTeamAction("Попытка удаления игрока из команды", player.getName(), teamName);
        if (!teamsConfig.contains(TEAMS_PATH + teamName)) {
            TeamMessageUtils.sendTeamMessage(player, Component.text("❌ Такой команды не существует!", NamedTextColor.RED));
            return;
        }

        List<String> members = getTeamMembers(teamName);
        if (!members.contains(player.getName())) {
            TeamMessageUtils.sendTeamMessage(player, Component.text("❌ Вы не состоите в этой команде!", NamedTextColor.RED));
            return;
        }

        String currentLeader = getTeamLeader(teamName);
        members.remove(player.getName());
        teamsConfig.set(TEAMS_PATH + teamName + MEMBERS_PATH, members);

        String prefix = getTeamPrefix(teamName);
        NamedTextColor teamColor = getTeamColor(teamName);
        Component prefixComponent = TeamUtils.createPrefixComponent(prefix, teamColor);

        if (player.getName().equals(currentLeader)) {
            if (members.isEmpty()) {
                teamsConfig.set(TEAMS_PATH + teamName, null);
                Component message = Component.text("✅ Команда ")
                        .append(prefixComponent)
                        .append(Component.text("[" + teamName + "] распущена, так как вы были единственным участником!", NamedTextColor.GREEN));
                TeamMessageUtils.sendTeamMessage(player, message);
            } else {
                String newLeader = members.get(0);
                teamsConfig.set(TEAMS_PATH + teamName + LEADER_PATH, newLeader);
                saveTeams();
                Component leaderMessage = Component.text("✅ Вы покинули команду ")
                        .append(prefixComponent)
                        .append(Component.text("[" + teamName + "], лидерство передано " + newLeader + "!", NamedTextColor.GREEN));
                TeamMessageUtils.sendTeamMessage(player, leaderMessage);
                Player newLeaderPlayer = plugin.getServer().getPlayer(newLeader);
                if (newLeaderPlayer != null) {
                    Component newLeaderMessage = Component.text("👑 Теперь вы лидер команды ")
                            .append(prefixComponent)
                            .append(Component.text("[" + teamName + "]!", NamedTextColor.GREEN));
                    TeamMessageUtils.sendTeamMessage(newLeaderPlayer, newLeaderMessage);
                }
                for (String memberName : members) {
                    Player member = plugin.getServer().getPlayer(memberName);
                    if (member != null && !member.equals(player) && !member.equals(newLeaderPlayer)) {
                        Component message = Component.text("⚡ Лидер команды ")
                                .append(prefixComponent)
                                .append(Component.text("[" + teamName + "] сменился на " + newLeader + "!", NamedTextColor.YELLOW));
                        TeamMessageUtils.sendTeamMessage(member, message);
                    }
                }
            }
        } else {
            Component playerMessage = Component.text("✅ Вы покинули команду ")
                    .append(prefixComponent)
                    .append(Component.text("[" + teamName + "]!", NamedTextColor.GREEN));
            TeamMessageUtils.sendTeamMessage(player, playerMessage);
            for (String memberName : members) {
                Player member = plugin.getServer().getPlayer(memberName);
                if (member != null && !member.equals(player)) {
                    Component message = Component.text("⚡ Игрок " + player.getName() + " покинул команду ")
                            .append(prefixComponent)
                            .append(Component.text("[" + teamName + "]!", NamedTextColor.YELLOW));
                    TeamMessageUtils.sendTeamMessage(member, message);
                }
            }
        }

        saveTeams();
        updatePlayerPrefixes(teamName);
        plugin.getServer().getPluginManager().callEvent(new TeamChatListener.PlayerPrefixUpdateEvent(player, null)); // Сброс префикса
        plugin.getLogger().info("Игрок " + player.getName() + " покинул команду " + teamName);
    }

                .append(Component.text("[" + teamName + "] лидером " + leader.getName() + "!", NamedTextColor.RED));

    @Override
    public void updatePlayerPrefixes(String teamName) {
        ((MyPurpurPlugin) plugin).debugTeamAction("Обновление префиксов для команды", null, teamName);
        String prefix = getTeamPrefix(teamName);
        NamedTextColor teamColor = getTeamColor(teamName);
        Component prefixComponent = TeamUtils.createPrefixComponent(prefix, teamColor);

        List<String> members = getTeamMembers(teamName);
        for (String memberName : members) {
            Player member = plugin.getServer().getPlayer(memberName);
            if (member != null) {
                plugin.getServer().getPluginManager().callEvent(new TeamChatListener.PlayerPrefixUpdateEvent(member, prefixComponent));
                member.playerListName(prefixComponent.append(Component.text(member.getName(), NamedTextColor.WHITE))); // Ник без цвета
                plugin.debug("Префикс обновлён для игрока " + memberName + " в команде " + teamName);
            } else {
                plugin.getLogger().warning("Игрок " + memberName + " не найден онлайн для команды " + teamName);
            }
        }
    }

    @Override
    public String getPlayerTeam(@NotNull Player player) {
        for (String teamName : getTeamNames()) {
            if (getTeamMembers(teamName).contains(player.getName())) {
                ((MyPurpurPlugin) plugin).debugTeamAction("Игрок состоит в команде", player.getName(), teamName);
                return teamName;
            }
        }
        ((MyPurpurPlugin) plugin).debugTeamAction("Игрок не состоит в команде", player.getName(), null);
        return null;
    }

    @Override
    public @NotNull List<String> getTeamMembers(String teamName) {

/**
 * Обработчик команды /menu для открытия меню игрокам.
 */
public class MenuCommand implements org.bukkit.command.CommandExecutor {

    private final JavaPlugin plugin;
    private final PluginConfig pluginConfig;

    public MenuCommand(@NotNull JavaPlugin plugin, @NotNull PluginConfig pluginConfig) {
        this.plugin = plugin;
        this.pluginConfig = pluginConfig;
    }

    @Override
    public boolean onCommand(@NotNull CommandSender sender, @NotNull Command command, @NotNull String label, @NotNull String[] args) {
        if (!(sender instanceof Player player)) {
            sender.sendMessage(Component.text("❌ Эту команду может использовать только игрок!", NamedTextColor.RED));
            return true;
        }
        openMenu(player);
        return true;
    }

    private void openMenu(@NotNull Player player) {
        Inventory menu = plugin.getServer().createInventory(null, 9, Component.text("Меню"));

        ItemStack exampleItem = new ItemStack(Material.DIAMOND);
        ItemMeta meta = exampleItem.getItemMeta();
        if (meta != null) {
            meta.displayName(Component.text("Пример предмета"));
            exampleItem.setItemMeta(meta);
        }

        menu.setItem(4, exampleItem);

        player.openInventory(menu);
        player.playSound(
            player.getLocation(),
            Sound.BLOCK_NOTE_BLOCK_PLING, // Используем конкретный звук вместо valueOf
            Sound.Source.MASTER,
            (float) pluginConfig.getMenuSoundVolume(),
            (float) pluginConfig.getMenuSoundPitch()
        );
        player.getWorld().spawnParticle(org.bukkit.Particle.valueOf(pluginConfig.getMenuParticleEffect().toUpperCase()), player.getLocation().add(0.5, 1.0, 0.5), 30, 0.5, 0.5, 0.5, 0.1);
    }

/**
 * Слушатель событий, связанных с командами и чатом игроков.
 */
public class TeamChatListener implements Listener {

    private final TeamService teamManager;

    public TeamChatListener(@NotNull TeamService teamManager) {
        this.teamManager = teamManager;
    }

    @EventHandler
    public void onPlayerJoin(@NotNull PlayerJoinEvent event) {
        Player player = event.getPlayer();
        updatePlayerPrefix(player);
        ((MyPurpurPlugin) teamManager.getPlugin()).debugTeamAction("Обновлён префикс для игрока при входе", player.getName(), null);
    }

    @EventHandler
    public void onPlayerQuit(@NotNull PlayerQuitEvent event) {
        Player player = event.getPlayer();
        ((MyPurpurPlugin) teamManager.getPlugin()).debugTeamAction("Игрок вышел", player.getName(), null);
    }

    @EventHandler
    public void onPlayerChat(@NotNull AsyncChatEvent event) {
        Player player = event.getPlayer();
        String teamName = teamManager.getPlayerTeam(player);
        ((MyPurpurPlugin) teamManager.getPlugin()).debugTeamAction("Обработка сообщения от игрока", player.getName(), teamName);

        if (teamName != null) {
            String prefix = teamManager.getTeamPrefix(teamName);
            NamedTextColor teamColor = teamManager.getTeamColor(teamName);
            Component prefixComponent = Component.text("[" + prefix + "] ", teamColor);

            event.renderer((source, sourceDisplayName, message, viewer) ->
                    prefixComponent
                            .append(Component.text(source.getName(), NamedTextColor.WHITE))
                            .append(Component.text(": "))
                            .append(message.color(NamedTextColor.WHITE)) // Дефолтный цвет для текста сообщения
            );
        } else {
            event.renderer((source, sourceDisplayName, message, viewer) ->
                    Component.text(source.getName(), NamedTextColor.WHITE)
                            .append(Component.text(": "))
                            .append(message.color(NamedTextColor.WHITE)) // Дефолтный цвет для текста сообщения
            );
        }
    }

    @EventHandler
    public void onPlayerPrefixUpdate(@NotNull TeamChatListener.PlayerPrefixUpdateEvent event) {
        Player player = event.getPlayer();
        Component prefix = event.getPrefix();
        ((MyPurpurPlugin) teamManager.getPlugin()).debugTeamAction("Обновление префикса для игрока", player.getName(), null);

        if (prefix != null) {
            player.playerListName(prefix.append(Component.text(player.getName(), NamedTextColor.WHITE)));
        } else {
            player.playerListName(Component.text(player.getName(), NamedTextColor.WHITE)); // Сброс префикса
        }
    }

    private void updatePlayerPrefix(@NotNull Player player) {
        String teamName = teamManager.getPlayerTeam(player);
        ((MyPurpurPlugin) teamManager.getPlugin()).debugTeamAction("Обновление префикса для игрока", player.getName(), teamName);

        if (teamName != null) {
            String prefix = teamManager.getTeamPrefix(teamName);
            NamedTextColor teamColor = teamManager.getTeamColor(teamName);
            Component prefixComponent = Component.text("[" + prefix + "] ", teamColor);
            player.playerListName(prefixComponent.append(Component.text(player.getName(), NamedTextColor.WHITE)));
        } else {
            player.playerListName(Component.text(player.getName(), NamedTextColor.WHITE)); // Явный сброс префикса
        }
    }

    /**
     * Внутренний класс для события обновления префикса игрока.
     */

        private final Player player;

        public PlayerPrefixUpdateEvent(@NotNull Player player, Component prefix) {
            this.player = player;
            this.prefix = prefix; // Убрана аннотация @NotNull, чтобы разрешить null
        }

        public @NotNull Player getPlayer() {
            return player;
        }

        @SuppressWarnings("unused")

        }

        }

        }
    }

import java.util.Objects;

/**
 * Реализация сервиса управления командами.
 */
public class TeamManager implements TeamService {

    private static final String TEAMS_PATH = "teams.";
    private static final String LEADER_PATH = ".leader";
    private static final String MEMBERS_PATH = ".members";
    private static final String PREFIX_PATH = ".prefix";
    private static final String COLOR_PATH = ".color";

    private final JavaPlugin plugin;
    private FileConfiguration teamsConfig;
    private File teamsFile;
    private final Map<String, List<String>> teamMembersCache;
    private final Map<String, String> teamPrefixCache;
    private final Map<String, NamedTextColor> teamColorCache;

    public TeamManager(@NotNull JavaPlugin plugin) {
        this.plugin = plugin;
        this.teamMembersCache = new HashMap<>();
        this.teamPrefixCache = new HashMap<>();
        this.teamColorCache = new HashMap<>();
        loadTeams();
    }

    private void loadTeams() {
        teamsFile = new File(plugin.getDataFolder(), "teams.yml");

        // Убеждаемся, что папка плагина существует
        if (!plugin.getDataFolder().exists()) {
            boolean mkdirsResult = plugin.getDataFolder().mkdirs();
            if (!mkdirsResult) {
                plugin.getLogger().severe("⚠ Не удалось создать папку плагина: " + plugin.getDataFolder().getAbsolutePath());
                return;
            }
            ((MyPurpurPlugin) plugin).debug("📂 Папка плагина создана: " + plugin.getDataFolder().getAbsolutePath());
        }

        if (!teamsFile.exists()) {
            try {
                if (teamsFile.createNewFile()) {
                    plugin.getLogger().info("📂 Файл teams.yml успешно создан!");
                }
            } catch (IOException e) {
                plugin.getLogger().severe("⚠ Ошибка создания teams.yml: " + e.getMessage());
            }
        }

        teamsConfig = YamlConfiguration.loadConfiguration(teamsFile);
        reloadCaches();
        ((MyPurpurPlugin) plugin).debug("📂 Файл teams.yml загружен!");
    }

    private void saveTeams() {
        try {
            teamsConfig.save(teamsFile);
            reloadCaches(); // Обновляем кэш после сохранения
            ((MyPurpurPlugin) plugin).debug("📂 Файл teams.yml успешно сохранён.");
        } catch (IOException e) {
            plugin.getLogger().severe("⚠ Ошибка сохранения teams.yml: " + e.getMessage());
        }
    }

    private void reloadCaches() {
        teamMembersCache.clear();
        teamPrefixCache.clear();
        teamColorCache.clear();

        List<String> teamNames = getTeamNamesFromConfig();
        for (String teamName : teamNames) {
            teamMembersCache.put(teamName, teamsConfig.getStringList(TEAMS_PATH + teamName + MEMBERS_PATH));
            teamPrefixCache.put(teamName, teamsConfig.getString(TEAMS_PATH + teamName + PREFIX_PATH, ""));
            String color = teamsConfig.getString(TEAMS_PATH + teamName + COLOR_PATH, "WHITE");
            teamColorCache.put(teamName, NamedTextColor.NAMES.value(color.toLowerCase()));
        }
    }

    private List<String> getTeamNamesFromConfig() {
        if (teamsConfig.getConfigurationSection(TEAMS_PATH) == null) {
            return new ArrayList<>();
        }
        return new ArrayList<>(teamsConfig.getConfigurationSection(TEAMS_PATH).getKeys(false));
    }

    @Override
    public void createTeam(String teamName, String prefix, String color, @NotNull Player leader) {
        ((MyPurpurPlugin) plugin).debugTeamAction("Попытка создания команды", leader.getName(), teamName);
        String existingTeam = getPlayerTeam(leader);
        if (existingTeam != null) {
            TeamMessageUtils.sendTeamMessage(leader, Component.text("❌ Вы не можете создать команду, так как уже состоите в команде " + existingTeam + "!", NamedTextColor.RED));
            return;
        }

        if (teamsConfig.contains(TEAMS_PATH + teamName)) {
            TeamMessageUtils.sendTeamMessage(leader, Component.text("❌ Команда уже существует!", NamedTextColor.RED));
            return;
        }

        for (String existingTeamName : getTeamNames()) {
            String existingPrefix = getTeamPrefix(existingTeamName);
            if (prefix.equals(existingPrefix)) {
                TeamMessageUtils.sendTeamMessage(leader, Component.text("❌ Префикс '" + prefix + "' уже используется другой командой!", NamedTextColor.RED));
                return;
            }
        }

        NamedTextColor teamColor;
        try {
            teamColor = NamedTextColor.NAMES.value(color.toLowerCase());
            if (teamColor == null) {
                TeamMessageUtils.sendTeamMessage(leader, Component.text("❌ Неверный цвет команды. Используйте, например: RED, BLUE, GREEN и т.д.", NamedTextColor.RED));
                return;
            }
        } catch (IllegalArgumentException e) {
            TeamMessageUtils.sendTeamMessage(leader, Component.text("❌ Неверный цвет команды. Используйте, например: RED, BLUE, GREEN и т.д.", NamedTextColor.RED));
            return;
        }

        for (String existingTeamName : getTeamNames()) {
            NamedTextColor existingColor = getTeamColor(existingTeamName);
            if (color.equalsIgnoreCase(existingColor.toString())) {
                TeamMessageUtils.sendTeamMessage(leader, Component.text("❌ Цвет '" + color + "' уже используется другой командой!", NamedTextColor.RED));
                return;
            }
        }

        teamsConfig.set(TEAMS_PATH + teamName + LEADER_PATH, leader.getName());
        teamsConfig.set(TEAMS_PATH + teamName + MEMBERS_PATH, List.of(leader.getName()));
        teamsConfig.set(TEAMS_PATH + teamName + PREFIX_PATH, prefix);
        teamsConfig.set(TEAMS_PATH + teamName + COLOR_PATH, color.toUpperCase());
        saveTeams();

        Component prefixComponent = TeamUtils.createPrefixComponent(prefix, teamColor);
        Component message = Component.text("✅ Команда ")
                .append(prefixComponent)
                .append(Component.text("[" + teamName + "] создана! Вы теперь лидер команды "))
                .append(prefixComponent)
                .append(Component.text("[" + teamName + "]!", NamedTextColor.GREEN));
        TeamMessageUtils.sendTeamMessage(leader, message);
        updatePlayerPrefixes(teamName);
        plugin.getLogger().info("Команда " + teamName + " успешно создана лидером " + leader.getName());
    }

    @Override
    public void addPlayerToTeam(String teamName, @NotNull Player player) {
        ((MyPurpurPlugin) plugin).debugTeamAction("Попытка добавления игрока в команду", player.getName(), teamName);
        if (!teamsConfig.contains(TEAMS_PATH + teamName)) {
            TeamMessageUtils.sendTeamMessage(player, Component.text("❌ Такой команды не существует!", NamedTextColor.RED));
            return;
        }

        String existingTeam = getPlayerTeam(player);
        if (existingTeam != null) {
            TeamMessageUtils.sendTeamMessage(player, Component.text("❌ Вы уже состоите в команде " + existingTeam + "!", NamedTextColor.RED));
            return;
        }

        for (String team : getTeamNames()) {
            String leader = getTeamLeader(team);
            if (player.getName().equals(leader)) {
                TeamMessageUtils.sendTeamMessage(player, Component.text("❌ Вы не можете вступить в другую команду, так как являетесь лидером команды " + team + "!", NamedTextColor.RED));
                return;
            }
        }

        List<String> members = getTeamMembers(teamName);
        if (members.contains(player.getName())) {
            TeamMessageUtils.sendTeamMessage(player, Component.text("❌ Вы уже в этой команде!", NamedTextColor.RED));
            return;
        }

        members.add(player.getName());
        teamsConfig.set(TEAMS_PATH + teamName + MEMBERS_PATH, members);
        saveTeams();

        String prefix = getTeamPrefix(teamName);
        NamedTextColor teamColor = getTeamColor(teamName);
        Component prefixComponent = TeamUtils.createPrefixComponent(prefix, teamColor);

        for (String memberName : members) {
            Player member = plugin.getServer().getPlayer(memberName);
            if (member != null && !member.equals(player)) {
                Component message = Component.text("✅ Игрок " + player.getName() + " вступил в команду ")
                        .append(prefixComponent)
                        .append(Component.text("[" + teamName + "]!", NamedTextColor.GREEN));
                TeamMessageUtils.sendTeamMessage(member, message);
            }
        }
        Component playerMessage = Component.text("✅ Вы вступили в команду ")
                .append(prefixComponent)
                .append(Component.text("[" + teamName + "]!", NamedTextColor.GREEN));
        TeamMessageUtils.sendTeamMessage(player, playerMessage);
        updatePlayerPrefixes(teamName);
        plugin.getLogger().info("Игрок " + player.getName() + " вступил в команду " + teamName);
    }

    @Override
    public void removePlayerFromTeam(String teamName, @NotNull Player player) {
        ((MyPurpurPlugin) plugin).debugTeamAction("Попытка удаления игрока из команды", player.getName(), teamName);
        if (!teamsConfig.contains(TEAMS_PATH + teamName)) {
            TeamMessageUtils.sendTeamMessage(player, Component.text("❌ Такой команды не существует!", NamedTextColor.RED));
            return;
        }

        List<String> members = getTeamMembers(teamName);
        if (!members.contains(player.getName())) {
            TeamMessageUtils.sendTeamMessage(player, Component.text("❌ Вы не состоите в этой команде!", NamedTextColor.RED));
            return;
        }

        String currentLeader = getTeamLeader(teamName);
        members.remove(player.getName());
        teamsConfig.set(TEAMS_PATH + teamName + MEMBERS_PATH, members);

        String prefix = getTeamPrefix(teamName);
        NamedTextColor teamColor = getTeamColor(teamName);
        Component prefixComponent = TeamUtils.createPrefixComponent(prefix, teamColor);

        if (player.getName().equals(currentLeader)) {
            if (members.isEmpty()) {
                teamsConfig.set(TEAMS_PATH + teamName, null);
                Component message = Component.text("✅ Команда ")
                        .append(prefixComponent)
                        .append(Component.text("[" + teamName + "] распущена, так как вы были единственным участником!", NamedTextColor.GREEN));
                TeamMessageUtils.sendTeamMessage(player, message);
            } else {
                String newLeader = members.getFirst(); // Используем getFirst() для Java 21+
                teamsConfig.set(TEAMS_PATH + teamName + LEADER_PATH, newLeader);
                saveTeams();
                Component leaderMessage = Component.text("✅ Вы покинули команду ")
                        .append(prefixComponent)
                        .append(Component.text("[" + teamName + "], лидерство передано " + newLeader + "!", NamedTextColor.GREEN));
                TeamMessageUtils.sendTeamMessage(player, leaderMessage);
                Player newLeaderPlayer = plugin.getServer().getPlayer(newLeader);
                if (newLeaderPlayer != null) {
                    Component newLeaderMessage = Component.text("👑 Теперь вы лидер команды ")
                            .append(prefixComponent)
                            .append(Component.text("[" + teamName + "]!", NamedTextColor.GREEN));
                    TeamMessageUtils.sendTeamMessage(newLeaderPlayer, newLeaderMessage);
                }
                for (String memberName : members) {
                    Player member = plugin.getServer().getPlayer(memberName);
                    if (member != null && !member.equals(player) && !member.equals(newLeaderPlayer)) {
                        Component message = Component.text("⚡ Лидер команды ")
                                .append(prefixComponent)
                                .append(Component.text("[" + teamName + "] сменился на " + newLeader + "!", NamedTextColor.YELLOW));
                        TeamMessageUtils.sendTeamMessage(member, message);
                    }
                }
            }
        } else {
            Component playerMessage = Component.text("✅ Вы покинули команду ")
                    .append(prefixComponent)
                    .append(Component.text("[" + teamName + "]!", NamedTextColor.GREEN));
            TeamMessageUtils.sendTeamMessage(player, playerMessage);
            for (String memberName : members) {
                Player member = plugin.getServer().getPlayer(memberName);
                if (member != null && !member.equals(player)) {
                    Component message = Component.text("⚡ Игрок " + player.getName() + " покинул команду ")
                            .append(prefixComponent)
                            .append(Component.text("[" + teamName + "]!", NamedTextColor.YELLOW));
                    TeamMessageUtils.sendTeamMessage(member, message);
                }
            }
        }

        saveTeams();
        updatePlayerPrefixes(teamName);
        plugin.getServer().getPluginManager().callEvent(new TeamChatListener.PlayerPrefixUpdateEvent(player, null)); // Сброс префикса
        plugin.getLogger().info("Игрок " + player.getName() + " покинул команду " + teamName);
    }

    @Override
    public void kickPlayerFromTeam(String teamName, @NotNull Player leader, @NotNull Player target) {
        ((MyPurpurPlugin) plugin).debugTeamAction("Попытка исключения игрока из команды", target.getName(), teamName);
        if (!teamsConfig.contains(TEAMS_PATH + teamName)) {
            TeamMessageUtils.sendTeamMessage(leader, Component.text("❌ Такой команды не существует!", NamedTextColor.RED));
            return;
        }

        String currentLeader = getTeamLeader(teamName);
        if (!leader.getName().equals(currentLeader)) {
            TeamMessageUtils.sendTeamMessage(leader, Component.text("❌ Только лидер может выгнать участника!", NamedTextColor.RED));
            return;
        }

        List<String> members = getTeamMembers(teamName);
        if (!members.contains(target.getName())) {
            TeamMessageUtils.sendTeamMessage(leader, Component.text("❌ Этот игрок не в вашей команде!", NamedTextColor.RED));
            return;
        }

        if (target.getName().equals(currentLeader)) {
            TeamMessageUtils.sendTeamMessage(leader, Component.text("❌ Вы не можете выгнать себя из команды!", NamedTextColor.RED));
            return;
        }

        String prefix = getTeamPrefix(teamName);
        NamedTextColor teamColor = getTeamColor(teamName);
        Component prefixComponent = TeamUtils.createPrefixComponent(prefix, teamColor);

        members.remove(target.getName());
        teamsConfig.set(TEAMS_PATH + teamName + MEMBERS_PATH, members);
        saveTeams();

        Component leaderMessage = Component.text("✅ Игрок " + target.getName() + " выгнан из команды ")
                .append(prefixComponent)
                .append(Component.text("[" + teamName + "]!", NamedTextColor.GREEN));
        TeamMessageUtils.sendTeamMessage(leader, leaderMessage);
        Component targetMessage = Component.text("❌ Вы были выгнаны из команды ")
                .append(prefixComponent)
                .append(Component.text("[" + teamName + "] лидером " + leader.getName() + "!", NamedTextColor.RED));
        TeamMessageUtils.sendTeamMessage(target, targetMessage);
        for (String memberName : members) {
            Player member = plugin.getServer().getPlayer(memberName);
            if (member != null && !member.equals(leader) && !member.equals(target)) {
                Component message = Component.text("⚡ Игрок " + target.getName() + " был исключён из команды ")
                        .append(prefixComponent)
                        .append(Component.text("[" + teamName + "] лидером " + leader.getName() + "!", NamedTextColor.YELLOW));
                TeamMessageUtils.sendTeamMessage(member, message);
            }
        }
        updatePlayerPrefixes(teamName);
        plugin.getServer().getPluginManager().callEvent(new TeamChatListener.PlayerPrefixUpdateEvent(target, null)); // Сброс префикса
        plugin.getLogger().info("Игрок " + target.getName() + " исключён из команды " + teamName + " лидером " + leader.getName());
    }

    @Override
    public void transferLeadership(String teamName, @NotNull Player leader, @NotNull Player newLeader) {
        ((MyPurpurPlugin) plugin).debugTeamAction("Попытка передачи лидерства в команде", leader.getName(), teamName);
        if (!teamsConfig.contains(TEAMS_PATH + teamName)) {
            TeamMessageUtils.sendTeamMessage(leader, Component.text("❌ Такой команды не существует!", NamedTextColor.RED));
            return;
        }

        String currentLeader = getTeamLeader(teamName);
        if (!Objects.equals(currentLeader, leader.getName())) {
            TeamMessageUtils.sendTeamMessage(leader, Component.text("❌ Только лидер может передавать лидерство!", NamedTextColor.RED));
            return;
        }

        if (leader.getName().equals(newLeader.getName())) {
            TeamMessageUtils.sendTeamMessage(leader, Component.text("❌ Вы не можете передать лидерство самому себе!", NamedTextColor.RED));
            return;
        }

        List<String> members = getTeamMembers(teamName);
        if (!members.contains(newLeader.getName())) {
            TeamMessageUtils.sendTeamMessage(leader, Component.text("❌ Этот игрок не в вашей команде!", NamedTextColor.RED));
            return;
        }

        String prefix = getTeamPrefix(teamName);
        NamedTextColor teamColor = getTeamColor(teamName);
        Component prefixComponent = TeamUtils.createPrefixComponent(prefix, teamColor);

        teamsConfig.set(TEAMS_PATH + teamName + LEADER_PATH, newLeader.getName());
        saveTeams();

        Component leaderMessage = Component.text("✅ Лидерство передано " + newLeader.getName() + " в команде ")
                .append(prefixComponent)
                .append(Component.text("[" + teamName + "]!", NamedTextColor.GREEN));
        TeamMessageUtils.sendTeamMessage(leader, leaderMessage);
        Component newLeaderMessage = Component.text("👑 Теперь вы лидер команды ")
                .append(prefixComponent)
                .append(Component.text("[" + teamName + "]!", NamedTextColor.GREEN));
        TeamMessageUtils.sendTeamMessage(newLeader, newLeaderMessage);
        for (String memberName : members) {
            Player member = plugin.getServer().getPlayer(memberName);
            if (member != null && !member.equals(leader) && !member.equals(newLeader)) {
                Component message = Component.text("⚡ Лидер команды ")
                        .append(prefixComponent)
                        .append(Component.text("[" + teamName + "] сменился на " + newLeader.getName() + "!", NamedTextColor.YELLOW));
                TeamMessageUtils.sendTeamMessage(member, message);
            }
        }
        updatePlayerPrefixes(teamName);
        plugin.getLogger().info("Лидерство в команде " + teamName + " передано от " + leader.getName() + " к " + newLeader.getName());
    }

    @Override
    public void disbandTeam(String teamName, @NotNull Player leader) {
        ((MyPurpurPlugin) plugin).debugTeamAction("Попытка распустить команду", leader.getName(), teamName);
        if (!teamsConfig.contains(TEAMS_PATH + teamName)) {
            TeamMessageUtils.sendTeamMessage(leader, Component.text("❌ Такой команды не существует!", NamedTextColor.RED));
            return;
        }

        String currentLeader = getTeamLeader(teamName);
        if (!leader.getName().equals(currentLeader)) {
            TeamMessageUtils.sendTeamMessage(leader, Component.text("❌ Только лидер может распустить команду!", NamedTextColor.RED));
            return;
        }

        List<String> members = getTeamMembers(teamName);
        String prefix = getTeamPrefix(teamName);
        NamedTextColor teamColor = getTeamColor(teamName);
        Component prefixComponent = TeamUtils.createPrefixComponent(prefix, teamColor);

        teamsConfig.set(TEAMS_PATH + teamName, null);
        saveTeams();

        Component leaderMessage = Component.text("✅ Команда ")
                .append(prefixComponent)
                .append(Component.text("[" + teamName + "] распущена!", NamedTextColor.GREEN));
        TeamMessageUtils.sendTeamMessage(leader, leaderMessage);
        for (String memberName : members) {
            Player member = plugin.getServer().getPlayer(memberName);
            if (member != null && !member.equals(leader)) {
                Component message = Component.text("❌ Команда ")
                        .append(prefixComponent)
                        .append(Component.text("[" + teamName + "] была распущена лидером " + leader.getName() + "!", NamedTextColor.RED));
                TeamMessageUtils.sendTeamMessage(member, message);
            }
        }
        for (String memberName : members) {
            Player member = plugin.getServer().getPlayer(memberName);
            if (member != null) {
                plugin.getServer().getPluginManager().callEvent(new TeamChatListener.PlayerPrefixUpdateEvent(member, null)); // Сброс префикса
            }
        }
        plugin.getLogger().info("Команда " + teamName + " распущена лидером " + leader.getName());
    }

    @Override
    public void renameTeam(String oldTeamName, String newTeamName, @NotNull Player leader) {
        ((MyPurpurPlugin) plugin).debugTeamAction("Попытка переименования команды", leader.getName(), oldTeamName);
        if (!teamsConfig.contains(TEAMS_PATH + oldTeamName)) {
            TeamMessageUtils.sendTeamMessage(leader, Component.text("❌ Такой команды не существует!", NamedTextColor.RED));
            return;
        }

        String currentLeader = getTeamLeader(oldTeamName);
        if (!leader.getName().equals(currentLeader)) {
            TeamMessageUtils.sendTeamMessage(leader, Component.text("❌ Только лидер может переименовать команду!", NamedTextColor.RED));
            return;
        }

        if (teamsConfig.contains(TEAMS_PATH + newTeamName)) {
            TeamMessageUtils.sendTeamMessage(leader, Component.text("❌ Команда с таким названием уже существует!", NamedTextColor.RED));
            return;
        }

        teamsConfig.set(TEAMS_PATH + newTeamName, teamsConfig.getConfigurationSection(TEAMS_PATH + oldTeamName));
        teamsConfig.set(TEAMS_PATH + oldTeamName, null);
        saveTeams();

        String prefix = getTeamPrefix(newTeamName);
        NamedTextColor teamColor = getTeamColor(newTeamName);
        Component prefixComponent = TeamUtils.createPrefixComponent(prefix, teamColor);
        Component leaderMessage = Component.text("✅ Команда переименована в ")
                .append(prefixComponent)
                .append(Component.text("[" + newTeamName + "]!", NamedTextColor.GREEN));
        TeamMessageUtils.sendTeamMessage(leader, leaderMessage);
        TeamUtils.notifyTeamMembers(newTeamName, this, prefixComponent.append(Component.text("[" + newTeamName + "]!", NamedTextColor.GREEN)));
        updatePlayerPrefixes(newTeamName);
        plugin.getLogger().info("Команда " + oldTeamName + " переименована в " + newTeamName + " лидером " + leader.getName());
    }

    @Override
    public void setTeamPrefix(String teamName, String newPrefix, @NotNull Player leader) {
        ((MyPurpurPlugin) plugin).debugTeamAction("Попытка изменения префикса команды", leader.getName(), teamName);
        if (!teamsConfig.contains(TEAMS_PATH + teamName)) {
            TeamMessageUtils.sendTeamMessage(leader, Component.text("❌ Такой команды не существует!", NamedTextColor.RED));
            return;
        }

        String currentLeader = getTeamLeader(teamName);
        if (!leader.getName().equals(currentLeader)) {
            TeamMessageUtils.sendTeamMessage(leader, Component.text("❌ Только лидер может изменить префикс команды!", NamedTextColor.RED));
            return;
        }

        for (String existingTeamName : getTeamNames()) {
            if (!existingTeamName.equals(teamName)) {
                String existingPrefix = getTeamPrefix(existingTeamName);
                if (newPrefix.equals(existingPrefix)) {
                    TeamMessageUtils.sendTeamMessage(leader, Component.text("❌ Префикс '" + newPrefix + "' уже используется другой командой!", NamedTextColor.RED));
                    return;
                }
            }
        }

        teamsConfig.set(TEAMS_PATH + teamName + PREFIX_PATH, newPrefix);
        saveTeams();

        NamedTextColor teamColor = getTeamColor(teamName);
        Component prefixComponent = TeamUtils.createPrefixComponent(newPrefix, teamColor);
        Component leaderMessage = Component.text("✅ Префикс команды ")
                .append(prefixComponent)
                .append(Component.text("[" + teamName + "] изменён!", NamedTextColor.GREEN));
        TeamMessageUtils.sendTeamMessage(leader, leaderMessage);
        TeamUtils.notifyTeamMembers(teamName, this, prefixComponent.append(Component.text("[" + teamName + "]!", NamedTextColor.GREEN)));
        updatePlayerPrefixes(teamName);
        plugin.getLogger().info("Префикс команды " + teamName + " изменён на " + newPrefix + " лидером " + leader.getName());
    }

    @Override
    public void setTeamColor(String teamName, String newColor, @NotNull Player leader) {
        ((MyPurpurPlugin) plugin).debugTeamAction("Попытка изменения цвета команды", leader.getName(), teamName);
        if (!teamsConfig.contains(TEAMS_PATH + teamName)) {
            TeamMessageUtils.sendTeamMessage(leader, Component.text("❌ Такой команды не существует!", NamedTextColor.RED));
            return;
        }

        String currentLeader = getTeamLeader(teamName);
        if (!leader.getName().equals(currentLeader)) {
            TeamMessageUtils.sendTeamMessage(leader, Component.text("❌ Только лидер может изменить цвет команды!", NamedTextColor.RED));
            return;
        }

        NamedTextColor colorEnum;
        try {
            colorEnum = NamedTextColor.NAMES.value(newColor.toLowerCase());
            if (colorEnum == null) {
                TeamMessageUtils.sendTeamMessage(leader, Component.text("❌ Неверный цвет команды. Используйте, например: RED, BLUE, GREEN и т.д.", NamedTextColor.RED));
                return;
            }
        } catch (IllegalArgumentException e) {
            TeamMessageUtils.sendTeamMessage(leader, Component.text("❌ Неверный цвет команды. Используйте, например: RED, BLUE, GREEN и т.д.", NamedTextColor.RED));
            return;
        }

        for (String existingTeamName : getTeamNames()) {
            if (!existingTeamName.equals(teamName)) {
                String existingColor = teamsConfig.getString(TEAMS_PATH + existingTeamName + COLOR_PATH, "WHITE");
                if (newColor.equalsIgnoreCase(existingColor)) {
                    TeamMessageUtils.sendTeamMessage(leader, Component.text("❌ Цвет '" + newColor + "' уже используется другой командой!", NamedTextColor.RED));
                    return;
                }
            }
        }

        teamsConfig.set(TEAMS_PATH + teamName + COLOR_PATH, newColor.toUpperCase());
        saveTeams();

        NamedTextColor teamColor = NamedTextColor.NAMES.value(newColor.toLowerCase());
        String prefix = getTeamPrefix(teamName);
        Component prefixComponent = TeamUtils.createPrefixComponent(prefix, teamColor);
        Component leaderMessage = Component.text("✅ Цвет команды ")
                .append(prefixComponent)
                .append(Component.text("[" + teamName + "] изменён на " + teamColor + "!", NamedTextColor.GREEN));
        TeamMessageUtils.sendTeamMessage(leader, leaderMessage);
        TeamUtils.notifyTeamMembers(teamName, this, prefixComponent.append(Component.text("[" + teamName + "] изменён на " + teamColor + "!", NamedTextColor.GREEN)));
        updatePlayerPrefixes(teamName);
        plugin.getLogger().info("Цвет команды " + teamName + " изменён на " + newColor + " лидером " + leader.getName());
    }

    @Override
    public void updatePlayerPrefixes(String teamName) {
        ((MyPurpurPlugin) plugin).debugTeamAction("Обновление префиксов для команды", null, teamName);
        String prefix = getTeamPrefix(teamName);
        NamedTextColor teamColor = getTeamColor(teamName);
        Component prefixComponent = TeamUtils.createPrefixComponent(prefix, teamColor);

        List<String> members = getTeamMembers(teamName);
        for (String memberName : members) {
            Player member = plugin.getServer().getPlayer(memberName);
            if (member != null) {
                plugin.getServer().getPluginManager().callEvent(new TeamChatListener.PlayerPrefixUpdateEvent(member, prefixComponent));
                member.playerListName(prefixComponent.append(Component.text(member.getName(), NamedTextColor.WHITE))); // Ник без цвета
                ((MyPurpurPlugin) plugin).debug("Префикс обновлён для игрока " + memberName + " в команде " + teamName);
            } else {
                plugin.getLogger().warning("Игрок " + memberName + " не найден онлайн для команды " + teamName);
            }
        }
    }

    @Override
    public String getPlayerTeam(@NotNull Player player) {
        for (String teamName : getTeamNames()) {
            if (getTeamMembers(teamName).contains(player.getName())) {
                ((MyPurpurPlugin) plugin).debugTeamAction("Игрок состоит в команде", player.getName(), teamName);
                return teamName;
            }
        }
        ((MyPurpurPlugin) plugin).debugTeamAction("Игрок не состоит в команде", player.getName(), null);
        return null;
    }

    @Override
    public @NotNull List<String> getTeamMembers(String teamName) {
        List<String> cachedMembers = teamMembersCache.get(teamName);
        if (cachedMembers != null) {
            ((MyPurpurPlugin) plugin).debug("Получен кэшированный список участников команды " + teamName + ": " + cachedMembers);
            return new ArrayList<>(cachedMembers);
        }
        plugin.getLogger().warning("Команда " + teamName + " не найдена в кэше");
        return new ArrayList<>();
    }

    @Override
    public @NotNull List<String> getTeamNames() {
        List<String> teamNames = new ArrayList<>(teamMembersCache.keySet());
        ((MyPurpurPlugin) plugin).debug("Получен список команд: " + teamNames);
        return teamNames;
    }

    @Override
    public String getTeamPrefix(String teamName) {
        String prefix = teamPrefixCache.get(teamName);
        if (prefix != null) {
            ((MyPurpurPlugin) plugin).debug("Префикс команды " + teamName + ": " + prefix);
            return prefix;
        }
        ((MyPurpurPlugin) plugin).debug("Префикс команды " + teamName + " не найден, возвращается пустая строка");
        return "";
    }

    @Override
    public @NotNull NamedTextColor getTeamColor(String teamName) {
        NamedTextColor color = teamColorCache.get(teamName);
        if (color != null) {
            ((MyPurpurPlugin) plugin).debug("Цвет команды " + teamName + ": " + color);
            return color;
        }
        ((MyPurpurPlugin) plugin).debug("Цвет команды " + teamName + " не найден, возвращается WHITE");
        return NamedTextColor.WHITE;
    }

    @Override
    public String getTeamLeader(String teamName) {
        return teamsConfig.getString(TEAMS_PATH + teamName + LEADER_PATH);
    }

    @Override
    public @NotNull JavaPlugin getPlugin() {
        return plugin;
    }
}

 * Обработчик команды /menu для открытия меню игрокам.
 */
public class MenuCommand implements org.bukkit.command.CommandExecutor {

    private final JavaPlugin plugin;
    private final PluginConfig pluginConfig;

    public MenuCommand(@NotNull JavaPlugin plugin, @NotNull PluginConfig pluginConfig) {
        this.plugin = plugin;
        this.pluginConfig = pluginConfig;
    }

    @Override
    public boolean onCommand(@NotNull CommandSender sender, @NotNull Command command, @NotNull String label, @NotNull String[] args) {
        if (!(sender instanceof Player player)) {
            sender.sendMessage(Component.text("❌ Эту команду может использовать только игрок!", NamedTextColor.RED));
            return true;
        }
        openMenu(player);
        return true;
    }

        ItemStack exampleItem = new ItemStack(Material.DIAMOND);
        ItemMeta meta = exampleItem.getItemMeta();
        if (meta != null) {
            meta.displayName(Component.text("Пример предмета"));
            exampleItem.setItemMeta(meta);
        }

        menu.setItem(4, exampleItem);

        player.openInventory(menu);
        player.playSound(
            player.getLocation(),
            Sound.BLOCK_NOTE_BLOCK_PLING, // Используем конкретный звук вместо valueOf
            SoundCategory.MASTER, // Используем SoundCategory
            (float) pluginConfig.getMenuSoundVolume(),
            (float) pluginConfig.getMenuSoundPitch()
        );
        player.getWorld().spawnParticle(org.bukkit.Particle.valueOf(pluginConfig.getMenuParticleEffect().toUpperCase()), player.getLocation().add(0.5, 1.0, 0.5), 30, 0.5, 0.5, 0.5, 0.1);
    }
}
